```python
def rows_to_dataset(
    rows: list[dict], allowed_labels: tuple[str, ...] | None = SENTIMENT_LABELS
) -> list[tuple[str, str]]:
    """Pure transform: raw row dicts -> clean (text, label) pairs.

    Each row is a dict with keys {"feedback_text", "joined_text", "corrected_value"}.
    Drops rows with no resolvable text. Label filtering:
      - allowed_labels is a tuple -> drop any row whose corrected_value is not a member
        (sentiment default: SENTIMENT_LABELS — byte-stable pre-change behavior).
      - allowed_labels is None -> dynamic vocab: accept every non-empty corrected_value
        (used by build_category_dataset; the org's own corrections ARE the label space).
    """
    dataset: list[tuple[str, str]] = []
    for row in rows:
        text = _resolve_text(row)
        if not text:
            continue
        label = _normalize_label(row.get("corrected_value"))
        if not label:
            continue
        if allowed_labels is not None and label not in allowed_labels:
            continue
        dataset.append((text, label))
    return dataset
```

Note: the pre-change code never explicitly dropped blank normalized labels (it relied on
`label not in SENTIMENT_LABELS` to reject `""` implicitly, since `""` is never in
`SENTIMENT_LABELS`). Adding an explicit `if not label: continue` before the `allowed_labels`
check is a **behavior-preserving refactor** for the sentiment default (an empty label was
already always dropped) and is required for the `allowed_labels=None` dynamic path (otherwise
blank `corrected_value` rows would leak in as label `""`). Verify with
`test_allowed_labels_none_accepts_all_nonempty_corrected_value`'s blank-string case.

```python
def fetch_correction_rows(org_id: int, db, *, correction_type: str) -> list[dict]:
    """DB seam: fetch raw correction rows of `correction_type` for org_id from AICorrection,
    left-joined to FeedbackItem (same org only) for the text fallback.

    Lazy `sqlalchemy` import so this module (and the package as a whole) stays
    importable without sqlalchemy installed — only this one function needs it.
    """
    from sqlalchemy import text

    query = text(
        """
        SELECT ac.feedback_text                       AS feedback_text,
               fi.text                                AS joined_text,
               ac.corrected_value                     AS corrected_value
        FROM ai_corrections ac
        LEFT JOIN feedback_items fi
          ON fi.id = ac.entity_id
         AND fi.organization_id = ac.organization_id
        WHERE ac.organization_id = :org_id
          AND ac.correction_type = :correction_type
          AND ac.signal = 'correction'
          AND ac.corrected_value IS NOT NULL
        """
    )
    result = db.execute(query, {"org_id": org_id, "correction_type": correction_type})
    return [
        {
            "feedback_text": row.feedback_text,
            "joined_text": row.joined_text,
            "corrected_value": row.corrected_value,
        }
        for row in result
    ]


def fetch_sentiment_correction_rows(org_id: int, db) -> list[dict]:
    """Thin alias — byte-stable pre-change behavior/name preserved for existing callers."""
    return fetch_correction_rows(org_id, db, correction_type="sentiment")


def build_sentiment_dataset(org_id: int, db) -> list[tuple[str, str]]:
    """Composition: fetch + transform, sentiment-scoped (fixed 3-class vocab)."""
    return rows_to_dataset(fetch_sentiment_correction_rows(org_id, db))


def build_category_dataset(org_id: int, db) -> list[tuple[str, str]]:
    """Composition: fetch + transform, category-scoped (dynamic vocab — allowed_labels=None
    so every distinct corrected_value the org has ever corrected TO becomes a class)."""
    return rows_to_dataset(
        fetch_correction_rows(org_id, db, correction_type="category"), allowed_labels=None
    )


def derive_labels(dataset: list[tuple[str, str]]) -> tuple[str, ...]:
    """Sorted unique labels present in a (text, label) dataset — the full dynamic label
    set for a category dataset. NOTE (fair-A/B contract, PRD critique #3): this returns the
    FULL set. Callers that need to score an incumbent fairly (e.g. the keyword categorizer,
    which can only emit its own built-in vocab) MUST intersect this result with that
    incumbent's emittable label set before passing it to evaluate(labels=...) — see
    worker-trainer aspect. Passing the unintersected full set to evaluate() when the
    incumbent cannot emit some of those labels structurally depresses the incumbent's
    macro-F1 and over-credits the challenger."""
    return tuple(sorted({label for _, label in dataset}))
```

def _category_row(label: str, i: int, vocab_words: dict) -> tuple[str, str]:
    w = vocab_words[label][i % len(vocab_words[label])]
    return (f"{w} ticket number {i} details", label)


def _category_balanced_dataset(n_per_class: int, vocab_words: dict) -> list[tuple[str, str]]:
    rows = []
    for i in range(n_per_class):
        for label in vocab_words:
            rows.append(_category_row(label, i, vocab_words))
    return rows


def _category_gold_label(text: str, vocab_words: dict) -> str:
    for label, words in vocab_words.items():
        if any(w in text for w in words):
            return label
    return next(iter(vocab_words))  # fallback, never leaves incumbent undefined
```

    be the intersection of derive_labels(dataset) with the incumbent's built-in vocab — a
    label the incumbent can never emit (a custom category) must be excluded from the eval
    label set, so the challenger gets no free credit for classes the baseline structurally
    cannot guess. This test proves the CONTRACT at the evaluate() call boundary using a
    stand-in built-in vocab (the real intersection is computed by worker-trainer against
    the real keyword categorizer's vocab; this package only needs to prove evaluate()
    behaves correctly when handed the already-intersected label set)."""
    dataset = _category_balanced_dataset(20, _CATEGORY_VOCAB)  # billing, ui_bug, custom_widget_bug
    full_labels = derive_labels(dataset)
    assert "custom_widget_bug" in full_labels

    incumbent_builtin_vocab = {"billing", "ui_bug"}  # stand-in for the keyword categorizer's vocab
    fair_labels = tuple(l for l in full_labels if l in incumbent_builtin_vocab)
    assert fair_labels == ("billing", "ui_bug")
    assert "custom_widget_bug" not in fair_labels  # the load-bearing assertion

    def incumbent_predict(text: str) -> str:
        # An incumbent that can ONLY ever emit built-in labels (never the custom one) —
        # mirrors the real keyword categorizer's structural limitation.
        return _category_gold_label(text, {k: _CATEGORY_VOCAB[k] for k in incumbent_builtin_vocab})

    result = evaluate(
        dataset, incumbent_predict, train_classifier,
        labels=fair_labels, min_labels=MIN_LABELS, holdout_frac=0.2,
        min_holdout=MIN_HOLDOUT, margin=MARGIN,
    )
    # Never raises despite dataset containing a 3rd class outside `labels` — rows whose
    # true label is outside `labels` are simply never counted in either confusion matrix
    # (see _confusion_for's out-of-vocab guard, evaluate.py:52-57), so the custom-class
    # rows are silently excluded from scoring rather than crashing or being misattributed.
    assert result.decision in ("promoted", "retained", "skipped")
    assert result.n <= sum(1 for _, l in dataset if l in fair_labels)
```
